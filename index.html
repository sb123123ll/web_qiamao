function initDeviceOrientation() { if (window.DeviceOrientationEvent) { window.addEventListener('deviceorientation', function(event) { if (!physicsEnabled) return; // 获取设备的倾斜角度 (beta: 前后倾斜, gamma: 左右倾斜) let beta = event.beta; // -180 到 180 let gamma = event.gamma; // -90 到 90 // 如果获取不到数据（例如桌面设备），则使用默认重力 if (beta === null || gamma === null) return; // 限制角度，避免过度反转 if (beta > 90) beta = 90; if (beta < -90) beta = -90; // 根据设备倾斜计算重力方向和大小 // 当设备平放时(beta=0, gamma=0)，重力为0 // 当设备垂直时(beta=90)，垂直重力最大 const maxGravity = 0.4; // 最大重力加速度 // 屏幕旋转适配 const orientation = window.orientation || 0; let gravityX = 0; let gravityY = 0; if (orientation === 0) { // 竖屏 gravityX = (gamma / 90) * maxGravity; gravityY = (beta / 90) * maxGravity; } else if (orientation === 90) { // 横屏(向左转) gravityX = (beta / 90) * maxGravity; gravityY = -(gamma / 90) * maxGravity; } else if (orientation === -90) { // 横屏(向右转) gravityX = -(beta / 90) * maxGravity; gravityY = (gamma / 90) * maxGravity; } else if (orientation === 180) { // 倒竖屏 gravityX = -(gamma / 90) * maxGravity; gravityY = -(beta / 90) * maxGravity; } // 如果重力太小（设备接近平放），给一个默认向下的微小重力 if (Math.abs(gravityX) < 0.05 && Math.abs(gravityY) < 0.05) { gravityY = 0.15; } physicsConfig.gravityX = gravityX; physicsConfig.gravityY = gravityY; }); } }
